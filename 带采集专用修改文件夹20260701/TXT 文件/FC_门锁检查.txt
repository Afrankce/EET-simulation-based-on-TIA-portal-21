FUNCTION "FC_门锁检查"

VAR_IN_OUT
    电梯 : "UDT_单梯状态";
END_VAR

VAR_TEMP
    循环索引 : Int;
END_VAR

// ==================================================================
//  门锁全部闭合 = 轿厢门锁 AND 所有楼层的层门锁
//  直接读 DB_系统配置 的楼层范围，不再需要 Input 参数
// ==================================================================
#电梯.门锁全部闭合 := #电梯.轿厢门锁信号;

FOR #循环索引 := "DB_系统配置".建筑最低楼层 
                 TO "DB_系统配置".建筑最高楼层 DO
    IF NOT #电梯.楼层门锁信号[#循环索引] THEN
        #电梯.门锁全部闭合 := FALSE;
    END_IF;
END_FOR;
